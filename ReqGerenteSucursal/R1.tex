
\section{Requerimentos Funcionales del gerente de sucursal}
\begin{problemas}[BlueViolet]{}
\problema{RGS1}{Dar de alta empleado. El gerente dará de alta al empleado ingresando su nombre, edad y puesto.}{RC1}{Gerente de Sucursal}{Mediano}

\problema{RGS2}{Dar de baja a empleado. El gerente podrá dar de baja un empleado por medio de su ID.}{RC1}{Gerente de Sucursal}{bajo}

\problema{RGS3}{Visualizar datos del empelado. El gerente visualizará los datos del empleado como su nombre, edad, puesto, salario y su fotografía.}{RC1}{Gerente de Sucursal}{Bajo}

\problema{RGS4}{Modificar datos de empleado. El gerente podrá modifica los datos del empleado como su nombre, edad o puesto.}{RC1}{Gerente de Sucursal}{Bajo}

\problema{RGS5}{Agregar pan al catálogo. El gerente podrá registrar un pan insertando su nombre, precio y una imagen de este.}{RC1}{Gerente de Sucursal}{Mediano}

\problema{RGS6}{Modificar pan al catálogo. El gerente podrá modificar los panes registrados en el catálogo como su fotografía, precio o nombre.}{RC1}{Gerente de Sucursal}{Mediano}

\problema{RGS7}{Eliminar pan del catálogo. El gerente podrá eliminar pan registrado en el catálogo por medio del ID.}{RC1}{Gerente de Sucursal}{Mediano}

\problema{RGS8}{Consultar ventas. EL gerente podrá consultar las ventas realizadas en la sucursal en cualquier momento del día.}{RC1}{Gerente de Sucursal}{Mediano}

\problema{RGS9}{Dar de baja venta. El gerente podrá cancelar ventas por medio del identificador de esta.}{RC1}{Gerente de Sucursal}{bajo}

\problema{RGS10}{Consultar contabilidad. El gerente podrá consulta la contabilidad de su sucursal.}{RC1}{Gerente de Sucursal}{Alto}

\problema{RGS11}{Consultar esquema de producción. El gerente podrá consultar el esquema de producción de la panadería para visualizar que pan se produjo.}{RC1}{Gerente de Sucursal}{Alto}

\problema{RG12}{Consultar esquema de demanda. El gerente podrá consultar el esquema de demanda.}{RC1}{Gerente de Sucursal}{Alto}

\end{problemas}
