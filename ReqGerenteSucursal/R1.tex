\section{Requerimentos del gerente de sucursal}
\begin{problemas}[BlueViolet]{Req}
\problema{1}{Dar de alta empleado}{El gerente dará de alta al empleado ingresando su nombre, edad y puesto.}{}

\problema{2}{Dar de baja a empleado.}{El gerente podrá dar de baja un empleado por medio de su ID.}{}

\problema{3}{Visualizar datos del empelado}{El gerente visualizará los datos del empleado como su nombre, edad, puesto, salario y su fotografía.}{}

\problema{4}{Modificar datos de empleado.}{El gerente podrá modifica los datos del empleado como su nombre, edad o puesto.}{}

\problema{5}{Agregar pan al catálogo}{El gerente podrá registrar un pan insertando su nombre, precio y una imagen de este.}{}

\problema{6}{Modificar pan al catálogo}{El gerente podrá modificar los panes registrados en el catálogo como su fotografía, precio o nombre.
}{}

\problema{7}{Eliminar pan del catálogo}{El gerente podrá eliminar pan registrado en el catálogo por medio del ID.}{}

\problema{8}{Consultar ventas}{EL gerente podrá consultar las ventas realizadas en la sucursal en cualquier momento del día.}{}

\problema{9}{Dar de baja venta.}{El gerente podrá cancelar ventas por medio del identificador de esta.}{}

\problema{10}{Consultar contabilidad}{El gerente podrá consulta la contabilidad de su sucursal.}{}


\problema{11}{Consultar esquema de producción}{El gerente podrá consultar el esquema de producción de la panadería para visualizar que pan se produjo.}{}

\problema{12}{Consultar esquema de demanda}{El gerente podrá consultar el esquema de demanda.}{}
\end{problemas}
